generator client {
  provider = "prisma-client-js"
  previewFeatures = ["multiSchema"] // enable multi-schema
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
  schemas  = ["public", "passprism"]
}

model User {
  id        String   @id @default(uuid())
  email     String   @unique @db.VarChar(255)
  username  String   @unique @db.VarChar(32)
  password  String
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  chats     Chat[]

  @@schema("passprism")
}

model Chat {
  id         String   @id @default(uuid())
  userId     String
  chatTitle  String   @map("chat_title")  // underscore column if desired
  createdAt  DateTime @default(now()) @map("created_at")
  updatedAt  DateTime @updatedAt @map("updated_at")

  user       User     @relation(fields: [userId], references: [id], onDelete: Cascade, onUpdate: Cascade)
  histories  History[]

  @@schema("passprism")
}

model History {
  id         Int      @id @default(autoincrement())
  sessionId  String    @map("session_id")
  message    Json
  createdAt  DateTime  @default(now()) @map("created_at")

  chat       Chat     @relation(fields: [sessionId], references: [id], onDelete: Cascade, onUpdate: Cascade)

  @@map("n8n_chat_histories")
  @@schema("public")
}
